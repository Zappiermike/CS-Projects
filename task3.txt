	li loop2, loopStop

	li elsestatement, ELSE

	sgt test exponent number0
	jeqz test loop2
	jeqz exponent loop2
	rem test exponent number2

	;;;;;;;;;;;;;;;;;;;;;;;;;;;;

ELSE:	;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
	;;else:				;;

	;;	exponent = exponent - 1;;
